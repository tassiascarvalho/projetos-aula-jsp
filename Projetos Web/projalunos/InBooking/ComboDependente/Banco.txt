CREATE TABLE estado(

id_estado serial primary key,
sigla_estado char(2),
nome_estado varchar(50)

);

CREATE TABLE cidade(

id_cidade serial primary key,
fk_estado int,
nome_cidade varchar(50),

constraint fk_estado foreign key (fk_estado) references estado

);
